function colcon_build_release () {
    colcon build $@ --cmake-args -DCMAKE_BUILD_TYPE=Release --catkin-skip-building-tests  --symlink-install
}

function colcon_build_release_ignore_error () {
    colcon build $@ --symlink-install --cmake-args -DCMAKE_BUILD_TYPE=Release -DCMAKE_CXX_FLAGS="-w" --continue-on-error
}
